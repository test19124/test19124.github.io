#include <bits/stdc++.h>
using namespace std;

int mod=1000007;
int ji[1000010];
int top;
int me[1000010],to[1000010];

int X[4]={0,1};
int Y[4]={1,0};

struct zb {
  int a[5][5];
  int ans;
};
queue<zb>q;

void out(int a[5][5]) {
  for(int i=1;i<=3;i++) {
    for(int j=1;j<=3;j++) {
      cout<<a[i][j];
    }
    cout<<endl;
  }
}

void outji() {
  cout<<top<<endl;
  for(int i=0;i<mod;i++) {
    if(ji[i]) {
      for(int j=ji[i];;j=to[j]) {
        cout<<me[j]<<"-->";
        if(to[j]==0) {
          break;
        }
      }
      cout<<ji[i]<<endl;
    }
  }
}

int copy(int &a,int b[5][5]) {
  a=0;
  for(int i=1;i<=3;i++) {
    for(int j=1;j<=3;j++) {
      a=a*10+b[i][j];
    }
  }
  return a;
}

bool find(int x) {
  for(int i=ji[x%mod];to[i]!=0;i=to[i]) {
    if(me[i]==x) {
      return true;
    }
  }
  return false;
}

void in(int x) {
  if(ji[x%mod]==0) {
    top++;
    ji[x%mod]=top;
    me[top]=x;
    return;
  }
  for(int i=ji[x%mod];;i=to[i]) {
    if(to[i]==0) {
      top++;
      to[i]=top;
      me[top]=x;
      return;
    }
  }
}

zb push(int a[5][5],int ans) {
  zb x;
  for(int i=1;i<=3;i++) {
    for(int j=1;j<=3;j++) {
      x.a[i][j]=a[i][j];
    }
  }
  x.ans=ans;
  q.push(x);
  return x;
}
